1. 防抖函数 ​

防抖函数通常有两种模式，一种是先倒计时，后执行，另一种是先执行，后倒计时。

  • 先倒计时，后执行 ​

javascript
export function debounce(func, delay) {
  let timer = null;
  return function (...args) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      func.apply(this, args);
    }, delay);
  };
}

export function debounce(func, delay) {
  let timer;

    if (timer) clearTimeout(timer);

    timer = setTimeout(() => {
      timer = null;

    if (callNow) func.apply(this, args);

}

export function debounce(func, delay, immediate = false) {
  let timer;

    if (immediate) {

}

2. 节流函数 ​

节流函数与防抖函数的不同在于，节流的主要目的不是防止代码执行，而是将频繁密集的代码按固定的时间间隔分开执行，不具有阻塞性。

同样节流函数有先倒计时，后执行和先执行，后倒计时两种形态。

  • 先倒计时，后执行 ​

javascript
export function throttle(func, wait) {
  let timeout;
  return function () {
    let context = this;
    let args = arguments;
    if (!timeout) {
      timeout = setTimeout(() => {
        timeout = null;
        func.apply(context, args);
      }, wait);
    }
  };
}

  • 先执行，后倒计时 ​

javascript
export function throttle(func, wait) {
  let previous = 0;
  return function () {
    let now = Date.now();
    let context = this;
    let args = arguments;
    if (now - previous > wait) {
      func.apply(context, args);
      previous = now;
    }
  };
}

export function throttle(func, wait, type) {
  if (type === 1) {
    let previous = 0;
  } else if (type === 2) {
    let timeout;
  }
  return function () {
    let context = this;
    let args = arguments;
    if (type === 1) {
      let now = Date.now();
      if (now - previous > wait) {
        func.apply(context, args);
        previous = now;
      }
    } else if (type === 2) {
      if (!timeout) {
        timeout = setTimeout(() => {
          timeout = null;
          func.apply(context, args);
        }, wait);
      }
    }
  };
}

获取年龄 ​

javascript
const getAge = (birthday) => {
  const TODAY = new Date();
  const TODAY_YEAR = TODAY.getFullYear();
  const TODAY_MILLISECOND = TODAY.getTime();
  const BIRTHDAY = new Date(birthday);
  const BIRTHDAY_YEAR = BIRTHDAY.getFullYear();
  const BIRTHDAY_MONTH = BIRTHDAY.getMonth();
  const BIRTHDAY_DATE = BIRTHDAY.getDate();
  const BIRTHDAY_MILLISECOND = BIRTHDAY.getTime();
  const CURRENT_YEAR_BIRTHDAY = new Date(
    TODAY_YEAR,
    BIRTHDAY_MONTH,
    BIRTHDAY_DATE
  );
  const CURRENT_YEAR_BIRTHDAY_MILLISECOND = CURRENT_YEAR_BIRTHDAY.getTime();
  const CROSS_TIME = CURRENT_YEAR_BIRTHDAY_MILLISECOND - BIRTHDAY_MILLISECOND;
  const REAL_CROSS_TIME = TODAY_MILLISECOND - BIRTHDAY_MILLISECOND;
  return CROSS_TIME < REAL_CROSS_TIME
    ? TODAY_YEAR - BIRTHDAY_YEAR + 1
    : TODAY_YEAR - BIRTHDAY_YEAR;
};

双叹号（!!） ​

!!等于Boolean()布尔包装器，可对任何值显示转换成布尔值，例如

javascript
const str = "123";
console.log(Boolean(str)); //true
console.log(!!str); //true

可选链（?.） ​

可选链操作符是 ES2019 的正式特性，其本身虽然说不上奇技淫巧，但是和空值合并操作符（??）合体使用将能造成非常高的阅读难度暴击值

过去我们把某个对象的深层属性赋值给变量，如果不对每一层数据是否存在做校验，JS 会报中止错误，阻止后续代码执行。所以通常人们要如下操作避免此类问题：

javascript
const foo = obj.bar && obj.bar.baba;
1

有了可选连操作符后，一切都变得简单了

javascript
const foo = obj.bar?.baba;
1

更多内容还是得看 MDN可选链操作符
