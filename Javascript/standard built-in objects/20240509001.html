Promise 静态方法中的 all() 和 race() ​

简述 ​

all()和race()的相同之处都是接收一个元素为 Promise 对象的可迭代类型（iterable）作为参数，并且返回一个 Promise 对象。

区别在于：

  • all()参数中所有 Promise 都完成或者有一个 reject 了，all()返回的 Promise 才会进入 fulfilled 状态，调用其自身的then()方法执行接下来的操作；
  • race()参数中任何一个 Promise 对象 resolve 或者 reject 都会立刻使race()返回的 Promise 进入 fulfilled 状态；

示例代码 ​

javascript
const ITERATION_COUNT = 10;

function badRequest() {
  return Math.floor(Math.random() * ITERATION_COUNT);
}

function returnRandomTime() {
  return Math.ceil(Math.random() * 1000);
}

function asyncRequest(index, isBad) {
  const DELAY_TIME = returnRandomTime();
  const result = { index, DELAY_TIME };
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (isBad) {
        reject(result);
      } else {
        resolve(result);
      }
    }, DELAY_TIME);
  });
}

const REQUEST_QUEUE = [];
const result = [];
const BAD_REQUEST = badRequest();

for (let i = 0; i < ITERATION_COUNT; i++) {
  REQUEST_QUEUE.push(
    asyncRequest(i, i === BAD_REQUEST ? true : false)
      .then((res) => {
        result.push(`${res.index}:${res.DELAY_TIME}`);
      })
      .catch((err) => {
        console.log("asyncRequest执行报错，信息为：", err);
      })
  );
}

Promise.all(REQUEST_QUEUE)
  .then((res) => {
    console.log(
      "Promise.all执行完毕，结果为：",
      res,
      ",result结果为：",
      result
    );
  })
  .catch((err) => {
    console.log("Promise.all报错，信息为：", err, result);
  })
  .finally((final) => {
    console.log("Promise.all执行finally，结果为：", result);
  });

Promise.race(REQUEST_QUEUE)
  .then((res) => {
    console.log(
      "Promise.race执行完毕，结果为：",
      res,
      ",result结果为：",
      result
    );
  })
  .catch((err) => {
    console.log("Promise.race报错，信息为：", err, result);
  })
  .finally((final) => {
    console.log("Promise.race执行finally，结果为：", result);
  });
