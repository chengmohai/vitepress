import type { ObjectConfig } from "@/types/config";

const init = Symbol("init");

/**
 * @描述 水印
 * @param {string} unitName 水印第一排文字，单位名称
 * @param {string} userNameTime 水印第二排文字，登录人名称及时间
 * @param {string} containerID 水印挂载的目标节点ID
 * @param {object} waterMarkConfig 水印文字配置参数
 */
class Watermark {
  public unitName: string;
  public userNameTime: string;
  public containerID: string;
  public waterMarkConfig: ObjectConfig<any>;

  constructor(
    unitName = "",
    userNameTime = "",
    containerID = "",
    waterMarkConfig = {
      font: "20px 黑体",
      fillStyle: "rgba(200, 200, 200, 0.30)",
      textAlign: "center",
      textBaseline: "middle",
    }
  ) {
    this.unitName = unitName;
    this.userNameTime = userNameTime;
    this.containerID = containerID;
    this.waterMarkConfig = waterMarkConfig;
  }

  [init]() {
    const mark = document.createElement("canvas");
    // 倾斜角
    const ANGLE = (-18 * Math.PI) / 180;
    const ANTI_ANGLE = (-(90 - 18) * Math.PI) / 180;
    // 副标题文本长度在有倾斜角的情况下，通过计算获取直角三角形底边的长度
    const UNITNAME_LENGTH = this.unitName.length * 20;
    const USERNAME_LENGTH = this.userNameTime.length * 20 + 20;
    // 主标题和副标题哪个长用哪个
    const LONGER_LENGTH =
      UNITNAME_LENGTH >= USERNAME_LENGTH ? UNITNAME_LENGTH : USERNAME_LENGTH;
    mark.width =
      LONGER_LENGTH * Math.cos(ANGLE) + 20 > 300
        ? LONGER_LENGTH * Math.cos(ANGLE) + 20
        : 300;
    mark.height = LONGER_LENGTH * Math.cos(ANTI_ANGLE) + 50;
    const context = mark.getContext("2d") as CanvasRenderingContext2D;
    // 改变canvas绘制坐标起点从左上角移动到图形中心
    context?.translate(mark.width / 2, mark.height / 2);
    context.rotate(ANGLE);
    context.font = this.waterMarkConfig.font;
    context.fillStyle = this.waterMarkConfig.fillStyle;
    context.textAlign = this.waterMarkConfig.textAlign;
    context.textBaseline = this.waterMarkConfig.textBaseline;
    context.fillText(this.unitName, 0, -20);
    context.fillText(this.userNameTime, 0, 20);
    const div = document.createElement("div");
    div.id = "watermark_wrap";
    div.style.pointerEvents = "none";
    div.style.top = "0px";
    div.style.left = "0px";
    div.style.position = "absolute";
    div.style.zIndex = "1";
    div.style.width = "100%";
    div.style.height = "100%";
    div.style.background = `url(${mark.toDataURL(
      "image/png"
    )}) left top repeat`;
    document.getElementById(this.containerID)?.appendChild(div);
  }

  render() {
    const el = document.getElementById(this.containerID);
    this[init]();
    const htmlElementObserver = new MutationObserver((lists) => {
      const { removedNodes } = lists[0];
      const removeWatermark = Array.from(removedNodes).some((elem) => {
        return (elem as HTMLElement).id === "watermark_wrap";
      });
      if (removeWatermark) {
        console.log("【警告】：请勿删除水印！");
        this[init]();
      }
    });
    if (el) htmlElementObserver.observe(el, { childList: true });
  }
}

export default Watermark;
