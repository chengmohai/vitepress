存储对象 ​

根据构造函数传入的 mode 类型，判断使用localStorage还是sessionStorage方式存储，默认使用 localStorage

javascript
export class Storage {
  constructor(mode = "localStorage") {
    this.mode = mode;
  }

  static modeHandle(mode) {
    return window[mode === "localStorage" ? "localStorage" : "sessionStorage"];
  }

  static paramCheck(type, data) {
    if (typeof data !== type) throw new Error(`数据类型不为${type}`);
  }

  setItem(name, data) {
    Storage.modeHandle(this.mode).setItem(
      name,
      typeof data === "string" ? data : JSON.stringify(data)
    );
  }

  getItem(name) {
    Storage.paramCheck("string", name);
    return Storage.modeHandle(this.mode).getItem(name);
  }

  removeItem(name) {
    Storage.paramCheck("string", name);
    Storage.modeHandle(this.mode).removeItem(name);
  }

  setToken(token) {
    Storage.modeHandle(this.mode).setItem(
      "token",
      typeof token === "string" ? token : JSON.stringify(token)
    );
  }

  getToken() {
    return Storage.modeHandle(this.mode).getItem("token");
  }

  removeToken() {
    Storage.modeHandle(this.mode).removeItem("token");
  }

  clear() {
    Storage.modeHandle(this.mode).clear();
  }
}
