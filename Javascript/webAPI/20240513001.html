2. 轮播图功能需求 ​

序号需求
1点击左右按钮能实现对应方向的移动一张图片
2页面加载后，轮播图按间隔时间开始自动播放
3鼠标移入轮播图自动播放停止
4鼠标移出轮播图继续自动播放
5指示器跟随轮播图改变激活的小圆点
6点击指示器，轮播图跳转到对应的图片

3. 轮播图的功能实现 ​

3.1 图片移动的基础功能

javascript
var i = 0, //初始化当前显示第几张图片；i的作用还包括1：控制向左移动的次数；2：指示具体是哪个标记小圆点处于激活状态
  LIWidth = 600, // 每个li的固定宽度
  Duration = 500, // 每次轮播动画持续的时间
  LICount = 8, // li的总个数 // 要移动的ul
  ulImgs = document.getElementById("ul-imgs"), // 包含小圆点列表的ul
  ulIdxs = document.getElementById("ul-idxs"), // 小圆点的元素列表
  lis = ulIdxs.children;
// 从当前位置移动到任意位置
function moveTo(to) {
  // 如果to没给值则等效于跳到 i+1，既下一张图；如果给了具体的数值，则 i=to
  if (to == undefined) {
    to = i + 1;
  } // 当轮播图在第一张图时
  if (i == 0) {
    if (to > i) {
      // 继续向右播放是不会出问题的
      ulImgs.className = "transition";
    } else {
      // 只有轮播图处于第一张图片，并且还要向左查看时才会出问题
      // 为了避免用户看到移动到最右边还是第一张图的障眼法露馅，先取消transition动画效果
      ulImgs.className = ""; // 瞬间将ul移动到最右边
      ulImgs.style.marginLeft = -LIWidth * LICount + "px"; // 100毫秒后挪动到真正要显示的图片
      setTimeout(() => {
        moveTo(LICount - 1);
      }, 100);
      return;
    }
  }
  i = to; // 先将表示第几张图片的变量i变为目标位置 // 再计算ulimgs的marginLeft距离
  ulImgs.style.marginLeft = -i * LIWidth + "px"; // 第一次循环先删掉所有小圆点的class
  for (var li of lis) {
    li.className = "";
  }
  if (i == LICount) {
    i = 0; // 当transition动画播放完毕清掉transition样式，将ul拉回到0位置的时候才不会有动画效果，防止露馅
    setTimeout(() => {
      ulImgs.className = "";
      ulImgs.style.marginLeft = 0;
    }, Duration);
  } // 再给当前位置的小圆点添加acitve的class
  lis[i].className = "active";
}

3.2 左右按钮

javascript
var btnLeft = document.getElementById("btn-left"),
  btnRight = document.getElementById("btn-right"), // 防抖开关，防止用户在短时间内快速点击按钮，导致效果出错
  canClick = true;
btnRight.onclick = function () {
  move(1);
};
btnLeft.onclick = function () {
  move(-1);
};
// 封装按钮公用函数
// n为1时表示向左移动，n为-1时表示向右移动
function move(n) {
  // 当canClick为true时才可以调用moveTo函数
  if (canClick) {
    moveTo(i + n); // 调用之后马上关闭canClick，防止再次点击
    canClick = false; // 在500毫秒后解封canClick
    setTimeout(() => {
      canClick = true;
    }, Duration);
  }
}

3.5 指示器功能

javascript
ulIdxs.onclick = function (e) {
  if (canClick) {
    var li = e.target;
    if (li.nodeName == "LI") {
      if (li.className !== "active") {
        for (var i = 0; i < lis.length; i++) {
          if (lis[i] == li) {
            break;
          }
        }
        moveTo(i);
        canClick = false;
        setTimeout(() => {
          canClick = true;
        }, Duration);
      }
    }
  }
};
