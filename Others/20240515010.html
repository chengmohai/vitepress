import nav from "./options/nav.js";
import sidebar from "./options/sidebar.js";
import defaultConfig from "./options/config.js";
import { defineConfig } from "vitepress";
import VueDevTools from "vite-plugin-vue-devtools";
import vueJsx from "@vitejs/plugin-vue-jsx";
import { resolve } from "node:path";
import { createSvgIconsPlugin } from "vite-plugin-svg-icons";
export default defineConfig({
  // 默认设置
  ...defaultConfig,
  vite: {
    plugins: [
      vueJsx(),
      VueDevTools(),
      createSvgIconsPlugin({
        iconDirs: [resolve(process.cwd(), "docs/.vitepress/svg/icons")],
        symbolId: "icon-[name]",
        inject: "body-last",
        customDomId: "__svg__icons__dom__",
      }),
    ],
  },
  markdown: {
    lineNumbers: true,
    codeCopyButtonTitle: "拷贝代码",
  },
  themeConfig: {
    // 文件存在srcDir/public/
    // https://vitepress.dev/zh/reference/default-theme-config#outline
    logo: "/logo.svg",
    outline: {
      label: "页面导航",
      level: [2, 3],
    },
    docFooter: {
      prev: "上一篇",
      next: "下一篇",
    },
    lastUpdated: {
      text: "上次更新于",
      formatOptions: {
        dateStyle: "long",
        timeStyle: "medium",
      },
    },
    search: {
      provider: "local",
    },
    // 顶部导航条
    nav,
    // 侧边栏
    sidebar,
    socialLinks: [
      { icon: "github", link: "https://github.com/chengmohai/vitepress" },
    ],
    returnToTopLabel: "回到顶部",
    sidebarMenuLabel: "菜单",
    darkModeSwitchLabel: "主题",
    lightModeSwitchTitle: "切换到浅色模式",
    darkModeSwitchTitle: "切换到深色模式",
  },
});

vue
// #region snippet
<script>
import * as echarts from "echarts";
export default {
  mounted() {
    // 也可以设置为 echarts.init(this.$refs.echarts, null, {width: "auto",height: 400,})
    // #region echarts-resize
    const chart = echarts.init(this.$refs.echarts);
    window.addEventListener("resize", () => {
      chart.resize();
    });
    // #endregion echarts-resize
    /** @type EChartsOption */
    chart.setOption({
      title: { text: "起个名" },
      tooltip: {},
      xAxis: {
        data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
      },
      yAxis: {},
      series: [
        {
          name: "销量",
          type: "bar",
          data: [5, 20, 36, 10, 10, 20],
        },
      ],
    });
  },
};
</script>
// #endregion snippet

<template>
  <div ref="echarts" class="echarts" />
</template>

<style scoped lang="scss">

  width: 100%;
  height: 400px;
}
