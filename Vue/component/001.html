1. 功能分析 ​

  • 单日期选择
  • 多日期选择
  • 区间选择 （重点）
  • 向上滑动的方式选择月，可以切换到不同的年份
  • 每周从周一开始还是从周日开始可选
  • 标记出今天
  • 今天之前的日期不可选
  • 多日期选择需要实现跨月功能

javascript
<template>
  <div class="content">
    <!-- 父组件负责接收子组件发送的日期对象，流程控制和数据存储 -->
    <!-- 日历根据每周第一天分为两种样式，第一天是周一为modeZH，第一天为周日的是modeUS -->
    <!-- 根据组件type属性切换不同的模式 -->
    <ul class="weekly">
      <li v-for="(day, index) of mode.mode" :key="index">
        {{ day }}
      </li>
    </ul>
    <!-- vuescroll页面滚动条组件 -->
    <vue-scroll :ops="ops">
      <!-- 每月日期组件 -->
      <Month
        ref="child"
        v-for="(item, index) of calendarArr"
        :key="index"
        v-bind:nowDate="{ item, index }"
        v-on:selectedDate="selectedDate"
      ></Month>
    </vue-scroll>
    <div class="btn-content">
      <div class="button" @click="reset">重置</div>
      <div class="button" @click="sendDate">确认</div>
    </div>
  </div>
</template>

<script>
import { Dialog } from "vant";
import Month from "./CalendarComponent/Month.vue";

export default {
  name: "Test",
  components: { Month },

  data() {
    return {
      // 点击事件数据，包括：点选次数，每次点击的具体日期对象，被选中日期存储数组
      clickEvent: { count: 0, index: [], chosenDate: [] },
      // vuescroll配置参数
      ops: {
        vuescroll: {
          mode: "native",
          sizeStrategy: "percent",
          detectResize: true
        },
        scrollPanel: {
          initialScrollY: false,
          initialScrollX: false,
          scrollingX: false,
          scrollingY: true,
          speed: 300,
          easing: undefined,
          verticalNativeBarPos: "right",
          maxHeight: undefined,
          maxWidth: undefined
        },
        rail: {
          keepShow: true
        },
        bar: {
          hoverStyle: true,
          onlyShowBarOnScroll: true, //是否只有滚动的时候才显示滚动条
          background: "#F5F5F5", //滚动条颜色
          opacity: 0.5, //滚动条透明度
          "overflow-x": "hidden"
        }
      },
      // 日历模式
      mode: { mode: [], computedDay: 0 },
      // 存储今日到明年年底的日期数组，数据结构往下翻
      calendarArr: [],
      crossMonth: false
    };
  },
  methods: {
    // 页面初始化
    pageInit() {
      let mode = {
        modeZH: ["一", "二", "三", "四", "五", "六", "日"],
        modeUS: ["日", "一", "二", "三", "四", "五", "六"]
      };
      this.$attrs.mode === "modeZH"
        ? (this.mode = { mode: mode.modeZH, computedDay: 6 })
        : (this.mode = { mode: mode.modeUS, computedDay: 0 });
      // 页面挂载后初始化当前日期为所有计算的源数据
      let { date, year, month } = this.$utils.dateInit();
      // 保存月份数组
      let calendarArr = [];
      // 按年循环
      for (let i = 0; i < 2; i++) {
        // 首年
        if (i === 0) {
          // 截止到本年年底还剩几个月，作为首年循环的循环条件
          for (let j = 0; j < 12 - month; j++) {
            calendarArr.push({
              rightNow: date,
              year: year + i,
              month: month + j,
              firstDay: new Date(year + i, month + j, 1),
              firstDayWeekly: new Date(year + i, month + j, 1).getDay(),
              lastDay: new Date(
                year + i,
                month + j,
                this.$utils.dayCountPerMonth(year + i, month + j)
              ),
              dayArray: this.dayArray(
                new Date(year + i, month + j, 1),
                new Date(year + i, month + j, 1).getDay(),
                new Date(
                  year + i,
                  month + j,
                  this.$utils.dayCountPerMonth(year + i, month + j)
                )
              )
            });
          }
        } else {
          for (let k = 0; k < 12; k++) {
            // 第二年开始的源数据
            let _date = new Date(year + i, k, 1);
            calendarArr.push({
              rightNow: date,
              year: year + i,
              month: k,
              firstDay: _date,
              firstDayWeekly: _date.getDay(),
              lastDay: new Date(
                year + i,
                k,
                this.$utils.dayCountPerMonth(year + i, k)
              ),
              dayArray: this.dayArray(
                new Date(year + i, k, 1),
                new Date(year + i, k, 1).getDay(),
                new Date(year + i, k, this.$utils.dayCountPerMonth(year + i, k))
              )
            });
          }
        }
      }
      this.calendarArr = calendarArr;
    },
    // 发送按钮函数
    sendDate() {
      // 判断当前组件的类型
      switch (this.$attrs.type) {
        case "range":
          // 如果只选了开始日期
          if (this.clickEvent.count === 1) {
            Dialog({
              title: "提示",
              message: "请选择截止日期"
            });
          } else {
            this.$emit("rangeDate", this.clickEvent);
          }
          break;
        case "single":
          this.$emit("singleDate", this.clickEvent);
          break;
        case "multiple":
          this.$emit("multipleDate", this.clickEvent);
      }
    },
    // 重置函数
    reset() {
      // 暴力重置整个日期数组
      for (let i = 0; i < this.calendarArr.length; i++) {
        for (let j = 0; j < this.calendarArr[i].dayArray.length; j++) {
          this.calendarArr[i].dayArray[j].actived = false;
          this.calendarArr[i].dayArray[j].between = false;
        }
      }
      switch (this.$attrs.type) {
        case "single":
          this.clickEvent.index = [];
          this.clickEvent.chosenDate = [];
          break;
        default:
          this.clickEvent = { count: 0, index: [], chosenDate: [] };
      }
    },
    // 子组件点击监控函数
    selectedDate(date) {
      // date为子组件向父组件传递的数据，包括：
      // 1、发送数据的日期数组的索引号
      // 2、完整的日期对象数据
      let { index: month, item: day, lastDay } = date;
      let ce = this.clickEvent;
      switch (this.$attrs.type) {
        case "range":
          // 首次点选
          if (ce.count === 0) {
            this.calendarArr[month].dayArray[day.index].actived = true;
            ce.count++;
            ce.index[0] = { monthIndex: month, day, lastDay };
            ce.chosenDate.push({ month, day });
            // 第二次点选
          } else if (ce.count === 1) {
            // 判断第二次点选日期是否在上一次之前
            // 如果是则结束日期选择操作，重置相关数据
            if (day.date.temDate < ce.index[0].day.date.temDate) {
              Dialog({
                title: "提示",
                message: "截止日期不可以晚于起始日期"
              });
              this.reset();
              // 正常的第二次点击
            } else {
              this.calendarArr[month].dayArray[day.index].actived = true;
              ce.count++;
              ce.index[1] = { monthIndex: month, day, lastDay };
              // 判断本次日期选择是否跨月
              if (ce.index[0].monthIndex === ce.index[1].monthIndex) {
                // 如果没有跨月，则对两次选择中间的日期对象进行操作
                for (
                  let i = ce.index[0].day.index + 1,
                    monthIndex = ce.index[0].monthIndex;
                  i < ce.index[1].day.index;
                  i++
                ) {
                  this.calendarArr[monthIndex].dayArray[i].between = true;
                  // 点击事件的被选中日期存储数组的顺序是：
                  // 1、先存处于between状态的日期对象
                  // 2、再存本次点击的日期对象
                  ce.chosenDate.push({
                    month,
                    day: this.calendarArr[monthIndex].dayArray[i]
                  });
                }
                ce.chosenDate.push({ month, day });
              } else {
                // 如果跨月需要对三种对象进行处理
                // 1.第一个月从选中之日到月底需要变成被选状态
                for (
                  let i = ce.index[0].day.index + 1,
                    monthIndex = ce.index[0].monthIndex;
                  i <=
                  ce.index[0].lastDay.getDate() -
                    ce.index[0].day.day +
                    ce.index[0].day.index;
                  i++
                ) {
                  this.calendarArr[ce.index[0].monthIndex].dayArray[
                    i
                  ].between = true;
                  ce.chosenDate.push({
                    month: monthIndex,
                    day: this.calendarArr[monthIndex].dayArray[i]
                  });
                }
                // 2.如果两个选择中间跨度超过一个月，则将所有整月变成被选状态
                for (
                  let k = ce.index[0].monthIndex + 1;
                  k < ce.index[1].monthIndex;
                  k++
                ) {
                  for (let h = 0; h < 42; h++) {
                    if (this.calendarArr[k].dayArray[h].visiable) {
                      this.calendarArr[k].dayArray[h].between = true;
                      ce.chosenDate.push({
                        month: k,
                        day: this.calendarArr[k].dayArray[h]
                      });
                    }
                  }
                }
                // 3.最后一个月从月初到截止日变成被选状态
                for (
                  let j = ce.index[1].day.index - ce.index[1].day.day,
                    monthIndex = ce.index[1].monthIndex;
                  j < ce.index[1].day.index;
                  j++
                ) {
                  this.calendarArr[monthIndex].dayArray[j].between = true;
                  ce.chosenDate.push({
                    month: monthIndex,
                    day: this.calendarArr[monthIndex].dayArray[j]
                  });
                }
              }
            }
            // 第三次点击时结束日期选择操作，重置相关数据
          } else if (ce.count === 2) {
            this.reset();
          }
          break;
        case "single":
          // 先清空所有选择
          this.reset();
          this.calendarArr[month].dayArray[day.index].actived = true;
          ce.index[0] = { monthIndex: month, day, lastDay };
          ce.chosenDate.push({ month, day });
          break;
        case "multiple":
          this.calendarArr[month].dayArray[day.index].actived = true;
          ce.count++;
          ce.index.push({ monthIndex: month, day, lastDay });
          ce.chosenDate.push({ month, day });
          break;
      }
    },
    dayArray(firstDay, firstDayWeekly, lastDay) {
      // 临时数组，保存待返回的当月每日对象
      let temArray = [],
        startTime;
      if (this.$attrs.mode === "modeZH") {
        // 本页面起始日期
        startTime =
          // 当前算法为中式，每周第一天为周一
          // 假如本月第一天是周二，则当月第一周排序为0的是周一
          // 算法：当月第一天 - （当月第一天在周几-1）
          firstDay -
          // 周日是0，第一天如果是周日，则需要减去前6天
          // computedDay为需要减去的天数
          (firstDayWeekly - 1 < 0
            ? this.mode.computedDay
            : firstDayWeekly - 1) *
            24 *
            60 *
            60 *
            1000;
      } else {
        startTime = firstDay - firstDayWeekly * 24 * 60 * 60 * 1000;
      }
      // 循环条件42表示每行7天，渲染6行。不建议渲染5行，有些大月挤挤插插的可能会造成错误显示
      for (let i = 0; i < 42; i++) {
        // 临时对象，保存当前循环的日期对象
        let temDate = new Date(startTime + i * 24 * 60 * 60 * 1000);
        // 判断，如果当前循环日期在本月第一天以前或最后一天以后，则其不可以显示
        temArray.push({
          index: i,
          date: { temDate, todayCheck: this.$utils.isToday(temDate) },
          year: temDate.getFullYear(),
          month: temDate.getMonth(),
          day: temDate.getDate(),
          weekly: temDate.getDay(),
          // 就是这
          visiable: temDate < firstDay || temDate > lastDay ? false : true,
          actived: false,
          between: false
        });
      }
      // 返回该数组
      return temArray;
    }
  },
  created() {
    this.pageInit();
  }
};
</script>

<template>
  <div class="month-content">
    <!-- Month组件主要用来渲染页面和向父组件发送被点选的日期对象 -->
    <!-- 年份提示 - 背景 -->
    <div class="bg">
      {{ nowDate.item.year }}
    </div>
    <!-- 月份提示 -->
    <div class="month-info">{{ calendarOps.monthZh[nowDate.item.month] }}</div>
    <!-- 循环渲染日期 -->
    <!-- style中visibility表示非当月日期处于隐藏状态 -->
    <!-- cursor和color表示当月已过日期不可点状态 -->
    <ul class="month-ul">
      <li
        class="month-li"
        v-for="(item, index) of nowDate.item.dayArray"
        :key="index"
        :style="{
          visibility: item.visiable ? '' : 'hidden',
          cursor:
            item.date.temDate < nowDate.item.rightNow - 24 * 60 * 60 * 1000
              ? 'not-allowed'
              : '',
          color:
            item.date.temDate < nowDate.item.rightNow - 24 * 60 * 60 * 1000
              ? '#cccccc'
              : ''
        }"
        :class="{
          actived: item.actived,
          between: item.between,
          today: item.date.todayCheck
        }"
        @click="selectDate(item, nowDate.index, nowDate.item.lastDay, $event)"
      >
        {{ item.day }}
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  name: "Month",
  props: ["nowDate"],
  data() {
    return {
      // 日期计算基础数据
      calendarOps: {
        // 月份
        // 中文背景
        monthZh: [
          "一月",
          "二月",
          "三月",
          "四月",
          "五月",
          "六月",
          "七月",
          "八月",
          "九月",
          "十月",
          "十一月",
          "十二月"
        ],
        // 英文背景
        monthEn: [
          "JAN",
          "FEB",
          "MAR",
          "APR",
          "MAY",
          "JUN",
          "JUL",
          "AUG",
          "SEP",
          "OCT",
          "NOV",
          "DEC"
        ]
      }
    };
  },
  methods: {
    selectDate(item, index, lastDay, e) {
      if (e.target.style.cursor !== "not-allowed") {
        this.$emit("selectedDate", { item, index, lastDay });
      }
    }
  }
};
</script>

module.exports = {
  // 封装闰年检查函数
  isLeapYear(year) {
    return (year % 4 == 0 && year % 100 == 0) || year % 400 == 0 ? true : false;
  },
  // 封装获取随机数函数
  getRandomNum(maxNum) {
    return Math.ceil(Math.random() * maxNum);
  },
  dayCountPerMonth(year, month) {
    var commonYear = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31],
      leapYear = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
    if (this.isLeapYear(year)) {
      return leapYear[month];
    } else {
      return commonYear[month];
    }
  },
  dateInit() {
    let _date = new Date();
    return {
      date: _date,
      year: _date.getFullYear(),
      month: _date.getMonth(),
    };
  },
  isToday(date) {
    var today = new Date();
    return today - date > 0 && today - date < 24 * 60 * 60 * 1000
      ? true
      : false;
  },
};
