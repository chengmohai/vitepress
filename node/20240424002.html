// /docs/.vitepress/utils/tools.js
import path from "node:path";
import fs from "node:fs";
import defaultConfig from "../options/config.js";
// æ–‡ä»¶æ ¹ç›®å½•
const BASE_URL = path.resolve("docs", defaultConfig.srcDir);
// åˆ¤æ–­æ˜¯å¦æ˜¯æ–‡ä»¶å¤¹
const isDirectory = (path) => fs.lstatSync(path).isDirectory();
// åˆ¤æ–­ç›®å½•ä¸­æ˜¯å¦å­˜åœ¨MDæ–‡ä»¶
const HAS_MD = (dir) => dir.some((item) => item.search(/\.md$/) !== -1);
// åˆ¤æ–­ç›®å½•ä¸­æ˜¯å¦æœ‰JSONé…ç½®æ–‡ä»¶
const HAS_JSON = (dir) => dir.some((item) => item.search(/\.json$/) !== -1);
// æ‰¾åˆ°JSONæ–‡ä»¶
const JSON_FILE = (arr, dir) => {
  const FILE_NAME = arr.find((item) => item.search(/\.json$/) !== -1);
  const FILE = fs.readFileSync(`${dir}/${FILE_NAME}`, "utf-8");
  const META_OBJ = JSON.parse(FILE.toString());
  return META_OBJ;
};
// è¿‡æ»¤åå•
const FILTER_LIST = ["index.md"];
// ç”Ÿæˆä¾§è¾¹æ 
function getList(name) {
  const res = [];
  const DIR_PATH = `${BASE_URL}/${name}`;
  const DIR = fs.readdirSync(DIR_PATH);
  for (let fileOrDir of DIR) {
    //  æ‹¼æ¥ç›®å½•
    const dir = path.join(DIR_PATH, fileOrDir);
    const isDir = isDirectory(dir);
    if (isDir) {
      let text = fileOrDir;
      let sort;
      const dirDetail = fs.readdirSync(dir);
      if (!dirDetail.length || !HAS_MD(dirDetail)) continue;
      if (HAS_JSON(dirDetail)) {
        const res = JSON_FILE(dirDetail, dir);
        if (res.hidden) continue;
        text = res.title;
        sort = res.sort;
      }
      res.push({
        text,
        sort,
        collapsible: true,
        items: generateSidebar(`${name}/${fileOrDir}`),
      });
    } else if (!FILTER_LIST.includes(fileOrDir)) {
      const extName = path.extname(`${DIR_PATH}/${fileOrDir}`);
      if (extName !== ".md") {
        continue;
      }
      const file = fs.readFileSync(`${dir}`, "utf-8");
      const text = file.match(/^#\s+.+/)[0].slice(2);
      res.push({
        text,
        link: `/${name}/${fileOrDir}`,
      });
    }
  }
  return res;
}
// ç”Ÿæˆä¾§è¾¹æ 
export function generateSidebar(name) {
  const LIST = getList(name);
  LIST.sort((a, b) => a.sort - b.sort);
  return LIST;
}
